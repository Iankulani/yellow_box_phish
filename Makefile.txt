.PHONY: help install test run clean docker-build docker-run format lint security

help:
	@echo "YellowBoxPhish Makefile"
	@echo ""
	@echo "  install      - Install dependencies"
	@echo "  test         - Run tests"
	@echo "  run          - Run application"
	@echo "  clean        - Clean temporary files"
	@echo "  docker-build - Build Docker image"
	@echo "  docker-run   - Run Docker container"
	@echo "  format       - Format code with black"
	@echo "  lint         - Run linters"
	@echo "  security     - Run security checks"

install:
	@echo "📦 Installing dependencies..."
	python3 -m venv venv
	. venv/bin/activate && pip install --upgrade pip
	. venv/bin/activate && pip install -r requirements.txt
	. venv/bin/activate && pip install -e .
	@echo "✅ Installation complete"

test:
	@echo "🧪 Running tests..."
	. venv/bin/activate && python requirements_check.py
	. venv/bin/activate && python test_commands.py

run:
	@echo "🚀 Starting YellowBoxPhish..."
	. venv/bin/activate && python yellow_box_phish.py

clean:
	@echo "🧹 Cleaning..."
	find . -type d -name "__pycache__" -exec rm -rf {} + 2>/dev/null || true
	find . -type d -name "*.egg-info" -exec rm -rf {} + 2>/dev/null || true
	find . -type f -name "*.pyc" -delete 2>/dev/null || true
	rm -rf .pytest_cache/ .mypy_cache/ .coverage htmlcov/ dist/ build/ 2>/dev/null || true
	@echo "✅ Clean complete"

docker-build:
	@echo "🐳 Building Docker image..."
	docker build -t yellowbox:latest .

docker-run:
	@echo "🐳 Running Docker container..."
	docker-compose up -d
	@echo "🌐 Web interface: http://localhost:8080"

docker-stop:
	@echo "🐳 Stopping Docker container..."
	docker-compose down

format:
	@echo "🎨 Formatting code..."
	. venv/bin/activate && black yellow_box_phish.py test_commands.py requirements_check.py

lint:
	@echo "🔍 Running linters..."
	. venv/bin/activate && flake8 yellow_box_phish.py --max-line-length=120
	. venv/bin/activate && mypy yellow_box_phish.py --ignore-missing-imports

security:
	@echo "🔒 Running security checks..."
	. venv/bin/activate && bandit -r . -f html -o bandit-report.html
	. venv/bin/activate && safety check -r requirements.txt
	@echo "📊 Security report: bandit-report.html"

all: install test run
